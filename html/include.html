Header & Footer

  • 여러 페이지에 공통적으로 들어가야하는 영역은 include/ 폴더에 html파일로 따로 저장.
    ajax를 이용하여 불러온 뒤, promise로 이후 동작 실행.
    promise는 IE에서 작동하지 않으므로 bluebird.js를 함께 이용.
  • include file을 불러온 뒤, html 파일 내 .header-area, .footer-area에 붙여넣음.

Source

win[namespace] = {
  navLoad: function () {
    (function () {
      return new Promise(function (resolve, reject) {
        $.get('/profile/html/include/header.html', function (response) {
          if (response) {
            resolve(response);
          }
          reject(new Error('Request is failed'));
        });
      });
    })()
    .then(function (data) {
      $('.header-area').html(data);
      // header load 후 evt
      win[namespace].nav.hoverMenu(); // hover evt on nav
      win[namespace].nav.slidingMenu(); // show/hide evt on nav
      win[namespace].nav.openDepth2(); // 2depth links evt on nav

    }).catch(function (err) {
      console.error('win.' + namespace + '.navLoad failed!!');
    });
  },
  footerLoad: function () {
    (function () {
      return new Promise(function (resolve, reject) {
        $.get('/profile/html/include/footer.html', function (response) {
          if (response) {
            resolve(response);
          }
          reject(new Error('Request is failed'));
        });
      });
    })().then(function (data) {
      $('.footer-area').html(data);
    }).catch(function (err) {
      console.error('win.' + namespace + '.footerLoad failed!!');
    });
  },
}
